def load_todos():
    """Load todos from JSON file"""
    if not os.path.exists(TODO_STORAGE_FILE):
        return []
    
    try:
        with open(TODO_STORAGE_FILE, "r") as f:
            return json.load(f)

    with open(TODO_STORAGE_FILE, "w") as f:

    """Get the next available ID"""
    todos = load_todos()
    if not todos:
        return 1
    return max(todo["id"] for todo in todos) + 1

def get_todos():

    if not todo:

    return todo
