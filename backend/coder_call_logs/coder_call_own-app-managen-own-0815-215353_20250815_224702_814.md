    db_entity = Entity(**entity.dict())
    db.add(db_entity)
    db.commit()
    db.refresh(db_entity)
    return db_entity

@router.get("/entities/", response_model=List[EntityResponse])

    return db.query(Entity).offset(skip).limit(limit).all()
